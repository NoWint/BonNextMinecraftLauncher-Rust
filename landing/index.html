// 并行下载队列 — 8 并发连接
let queue = DownloadQueue::new()
  .concurrency(8)
  .mirror(Mirror::AutoFailover)
  .verify(Verify::Sha1)
  .build();

// 实时进度推送
queue.on_progress(|event| {
  emit("download:progress", event);
});
